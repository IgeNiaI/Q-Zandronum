Command SpawnPlayer
	Player player with IndexTestOnly
	Byte priorState
	# Do we really need to send this? Shouldn't it always be PST_LIVE?
	Byte playerState
	Bool isBot
	Bool isSpectating
	Bool isDeadSpectator
	Bool isMorphed
	Long netid
	Angle angle
	Fixed x
	Fixed y
	Fixed z
	Byte playerClass

	If (isMorphed)
		Short morphStyle
		Class morphedClass
	EndIf
EndCommand

Command MovePlayer
	UnreliableCommand
	Player player with MoTest
	Bool isVisible

	# The server only sends position, angle, etc. information if the player is actually visible.
	If (isVisible)
		CheckFunction IsVisible
		ULong clientTicOnServerEnd
		# [BB] The x/y position has to be sent at full precision, otherwise the player may be rounded to a neighboring
		# sector on the clients, potentially completely changing its Z position.
		Fixed x
		Fixed y
		Fixed z
		Byte waterlevel
		Angle angle
		Fixed pitch
		Fixed velx
		Fixed vely
		Fixed velz
		Short ucmd_forwardmove
		Short ucmd_sidemove
		Short ucmd_upmove
		Fixed ucmd_yaw
		Fixed ucmd_pitch
		# Fixed ucmd_roll
		Long ucmd_buttons
	EndIf
EndCommand
