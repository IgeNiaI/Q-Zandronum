Command MoveThing
	Actor actor
	Short bits

	If (bits & CM_XY)
		Fixed newX
		Fixed newY
	EndIf

	If (bits & CM_Z)
		Fixed newZ
	EndIf

	If (bits & CM_ANGLE)
		Angle angle
	EndIf

	If (bits & CM_PITCH)
		Long pitch
	EndIf

	If (bits & CM_VELXY)
		Fixed velX
		Fixed velY
	EndIf

	If (bits & CM_VELZ)
		Fixed velZ
	EndIf

	If (bits & CM_MOVEDIR)
		Byte movedir
	EndIf

	If (bits & CM_LAST_XY)
		Fixed lastX
		Fixed lastY
	EndIf

	If (bits & CM_LAST_Z)
		Fixed lastZ
	EndIf
EndCommand
